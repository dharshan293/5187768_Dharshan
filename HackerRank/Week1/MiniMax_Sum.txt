#include <stdio.h>

int main() {
    long a[5], sum = 0, min, max;

    
    for (int i = 0; i < 5; i++) {
        scanf("%ld", &a[i]);
        sum += a[i];
    }

    min = max = a[0];

    
    for (int i = 1; i < 5; i++) {
        if (a[i] < min)
            min = a[i];
        if (a[i] > max)
            max = a[i];
    }

    
    printf("%ld %ld\n", sum - max, sum - min);

    return 0;
}
