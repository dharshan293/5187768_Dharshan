#include <stdio.h>
#include <stdlib.h>

int main() {
    int n;
    scanf("%d", &n);

    int arr[n][n];
    int primarySum = 0;
    int secondarySum = 0;

    // Reading the matrix and calculating sums
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &arr[i][j]);

            if (i == j)
                primarySum += arr[i][j];

            if (i + j == n - 1)
                secondarySum += arr[i][j];
        }
    }

    int difference = abs(primarySum - secondarySum);
    printf("%d\n", difference);

    return 0;
}
